# Stage 1: Build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /app

# Copy csproj and restore as distinct layers
COPY *.sln .
COPY IPDINProject.API/*.csproj ./IPDINProject.API/
COPY IPDINProject.Application/*.csproj ./IPDINProject.Application/
COPY IPDINProject.Domain/*.csproj ./IPDINProject.Domain/
COPY IPDINProject.Infrastructure/*.csproj ./IPDINProject.Infrastructure/
COPY IPDINProject.Presentation/*.csproj ./IPDINProject.Presentation/

RUN dotnet restore

# Copy the remaining files and build
COPY . .
WORKDIR /app/IPDINProject.API
RUN dotnet publish -c Release -o /out

# Stage 2: Runtime
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app
COPY --from=build /out .

# Expose port 80
ENV ASPNETCORE_URLS=http://+:80
ENV ASPNETCORE_ENVIRONMENT=Production

ENTRYPOINT ["dotnet", "IPDINProject.API.dll"]
